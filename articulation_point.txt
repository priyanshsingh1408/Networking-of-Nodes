class Solution {
private
    int timer = 1;
    void dfs(int node, int parent, vectorint &vis, int tin[], int low[],
             vectorint &mark, vectorintadj[]) {
        vis[node] = 1;
        tin[node] = low[node] = timer;
        timer++;
        int child = 0;
        for (auto it  adj[node]) {
            if (it == parent) continue;
            if (!vis[it]) {
                dfs(it, node, vis, tin, low, mark, adj);
                low[node] = min(low[node], low[it]);
                if (low[it] = tin[node] && parent != -1) {
                    mark[node] = 1;
                }
                child++;
            }
            else {
                low[node] = min(low[node], tin[it]);
            }
        }
        if (child  1 && parent == -1) {
            mark[node] = 1;
        }
    }
public
    vectorint articulationPoints(int n, vectorintadj[]) {
        vectorint vis(n, 0);
        int tin[n];
        int low[n];
        vectorint mark(n, 0);
        for (int i = 0; i  n; i++) {
            if (!vis[i]) {
                dfs(i, -1, vis, tin, low, mark, adj);
            }
        }
        vectorint ans;
        for (int i = 0; i  n; i++) {
            if (mark[i] == 1) {
                ans.push_back(i);
            }
        }
        if (ans.size() == 0) return { -1};
        return ans;
    }
};
int main() {

    int n = 5;
    vectorvectorint edges = {
        {0, 1}, {1, 4},
        {2, 4}, {2, 3}, {3, 4}
    };

    vectorint adj[n];
    for (auto it  edges) {
        int u = it[0], v = it[1];
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    Solution obj;
    vectorint nodes = obj.articulationPoints(n, adj);
    for (auto node  nodes) {
        cout  node   ;
    }
    cout  endl;
    return 0;

}
